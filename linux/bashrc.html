# Set GIT branch in prompt
parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1)/'
}

export PS1="\u@\h \W\[\033[32m\]\$(parse_git_branch)\[\033[00m\] $ "

alias tg=terragrunt
alias tf=terraform
alias grep='grep --color=auto'
alias ls='ls --color=auto'

export PATH=~/.local/bin:$PATH
